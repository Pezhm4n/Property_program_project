cmake_minimum_required(VERSION 3.20)
project(RealEstateCore VERSION 1.0.0 LANGUAGES C)

# Set C17 Standard
set(CMAKE_C_STANDARD 17)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Enable Warnings Policy based on Coding Standards
if(MSVC)
    add_compile_options(/W4 /WX) # /WX treats warnings as errors
    add_compile_definitions(_CRT_SECURE_NO_WARNINGS)
else()
    add_compile_options(-Wall -Wextra -Werror)
endif()

# Generate Migration Manifest at Configure Time (No Directory Traversal in C)
file(GLOB MIGRATION_FILES RELATIVE "${CMAKE_CURRENT_SOURCE_DIR}/migrations" "migrations/*.sql")
list(SORT MIGRATION_FILES)
set(MANIFEST_CONTENT "// Auto-generated by CMake\n#ifndef MIGRATION_MANIFEST_H\n#define MIGRATION_MANIFEST_H\n\nconst char* migration_files[] = {\n")
foreach(mig IN LISTS MIGRATION_FILES)
    string(APPEND MANIFEST_CONTENT "    \"${mig}\",\n")
endforeach()
string(APPEND MANIFEST_CONTENT "    0\n};\n\n#endif // MIGRATION_MANIFEST_H\n")
file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/generated/migration_manifest.h" "${MANIFEST_CONTENT}")

# Core Shared Library
add_library(re_core SHARED
    src/api/exports.c
    src/services/auth_service.c
    src/services/property_service.c
    src/services/report_service.c
    src/repository/db_connection.c
    src/repository/user_repo.c
    src/repository/property_repo.c
    src/repository/audit_repo.c
    src/repository/migrations.c
    src/dto/dto_validators.c
)

set_target_properties(re_core PROPERTIES PREFIX "" IMPORT_PREFIX "")

# Includes
target_include_directories(re_core PUBLIC include)
target_include_directories(re_core PUBLIC src/models)
target_include_directories(re_core PRIVATE src)
target_include_directories(re_core PRIVATE "${CMAKE_CURRENT_BINARY_DIR}/generated")
target_include_directories(re_core PRIVATE third_party/sqlite3)

# Third-party SQLite Integration (Strict Offline Build Rule)
if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/third_party/sqlite3/sqlite3.c")
    target_sources(re_core PRIVATE third_party/sqlite3/sqlite3.c)
else()
    find_package(unofficial-sqlite3 CONFIG QUIET)
    if(unofficial-sqlite3_FOUND)
        target_link_libraries(re_core PRIVATE unofficial::sqlite3::sqlite3)
    else()
        find_package(SQLite3 QUIET)
        if(SQLite3_FOUND)
            if(TARGET SQLite3::SQLite3)
                target_link_libraries(re_core PRIVATE SQLite3::SQLite3)
            else()
                target_include_directories(re_core PRIVATE ${SQLITE3_INCLUDE_DIRS})
                target_link_libraries(re_core PRIVATE ${SQLITE3_LIBRARIES})
            endif()
        else()
            # Fallback for MinGW/GCC environments resolving sqlite3 dynamically
            target_link_libraries(re_core PRIVATE sqlite3)
        endif()
    endif()
endif()

if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/third_party/cjson/cJSON.c")
    target_sources(re_core PRIVATE third_party/cjson/cJSON.c)
    target_include_directories(re_core PRIVATE third_party/cjson)
else()
    find_package(cJSON CONFIG QUIET)
    if(cJSON_FOUND)
        target_link_libraries(re_core PRIVATE cjson)
    else()
        find_path(CJSON_INCLUDE_DIR cjson/cJSON.h)
        find_library(CJSON_LIBRARY NAMES cjson)
        if(CJSON_INCLUDE_DIR AND CJSON_LIBRARY)
            target_include_directories(re_core PRIVATE ${CJSON_INCLUDE_DIR})
            target_link_libraries(re_core PRIVATE ${CJSON_LIBRARY})
        else()
            # Fallback for MinGW/GCC environments
            target_link_libraries(re_core PRIVATE cjson)
        endif()
    endif()
endif()

# Windows CryptoAPI for secure token generation (CryptGenRandom)
if(WIN32)
    target_link_libraries(re_core PRIVATE advapi32)
    # Statically link runtime libraries for MinGW to avoid MSYS2 DLL dependencies
    if(NOT MSVC)
        target_link_options(re_core PRIVATE -static-libgcc -static)
    endif()
endif()

# Optional: Enable testing
enable_testing()
